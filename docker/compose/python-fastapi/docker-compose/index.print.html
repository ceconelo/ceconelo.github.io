Docker Compose

docker-compose.yml

version: '3'
services:
  app:
    build:
      context: .
      dockerfile: ./.docker/dev/Dockerfile
    command:
      - /start
    container_name: python-fastapi
    env_file:
      - ./.docker/dev/.env
    ports:
      - 8000:8000
    volumes:
      - .:/home/python/app
      - ./.docker/dev/zsh/.p10k.zsh:/home/python/.p10k.zsh:delegated
      - ./.docker/dev/zsh/history:/home/python/zsh:delegated
